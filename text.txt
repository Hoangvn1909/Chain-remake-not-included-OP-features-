local gameStats = { alive = 0, killerName = "None", genCompleted = 0, totalGens = 0, fusePlaced = false, batteryInserted = false, timeLeft = 0, roundActive = false, ping = 0 }

local function updateGameStats()
    pcall(function()
        local playersFolder = Workspace:FindFirstChild("PLAYERS")
        if playersFolder then
            local aliveFolder = playersFolder:FindFirstChild("ALIVE")
            gameStats.alive = aliveFolder and #aliveFolder:GetChildren() or 0
            local killerFolder = playersFolder:FindFirstChild("KILLER")
            if killerFolder and #killerFolder:GetChildren() > 0 then
                gameStats.killerName = killerFolder:GetChildren()[1].Name
            else gameStats.killerName = "None" end
        end
        local maps = Workspace:FindFirstChild("MAPS")
        local gameMap = maps and maps:FindFirstChild("GAME MAP")
        if gameMap then
            local completed = 0; local total = 0
            for _, v in ipairs(gameMap:GetDescendants()) do
                if v.Name == "Generator" and v:IsA("Model") then
                    total = total + 1
                    local rep = v:FindFirstChild("Repair") or v:FindFirstChild("Progress")
                    if rep and ((rep:GetAttribute("Repaired") == true) or (rep:IsA("NumberValue") and rep.Value >= 100)) then completed = completed + 1 end
                end
            end
            gameStats.genCompleted = completed
            gameStats.totalGens = total > 0 and total or 4
        end
        local fuseBoxes = gameMap and gameMap:FindFirstChild("FuseBoxes")
        if fuseBoxes then
            local fusePlaced = false; local batteryInserted = false
            for _, box in ipairs(fuseBoxes:GetChildren()) do
                local fuse = box:FindFirstChild("Fuse")
                if fuse and fuse:GetAttribute("Placed") == true then fusePlaced = true end
                local battery = box:FindFirstChild("Battery")
                if battery and battery:GetAttribute("Inserted") == true then batteryInserted = true end
            end
            gameStats.fusePlaced = fusePlaced
            gameStats.batteryInserted = batteryInserted
        end
        local roundTimer = Workspace:FindFirstChild("RoundTimer")
        if roundTimer and roundTimer:IsA("NumberValue") then
            gameStats.timeLeft = roundTimer.Value
            gameStats.roundActive = true
        else
            local gameStuff = Workspace:FindFirstChild("GameStuff")
            local values = gameStuff and gameStuff:FindFirstChild("Values")
            if values then
                local time = values:GetAttribute("RoundTime")
                if time then gameStats.timeLeft = time; gameStats.roundActive = true else gameStats.roundActive = false end
            else gameStats.roundActive = false end
        end
        gameStats.ping = math.floor(LocalPlayer:GetNetworkPing() * 1000)
    end)
end

local function getSongTitle(soundId, callback)
    local http = game:GetService("HttpService")
    local url = "https://economy.roblox.com/v2/assets/" .. soundId .. "/details"
    local success, response = pcall(function() return http:GetAsync(url) end)
    if success and response then
        local data = http:JSONDecode(response)
        if data and data.Name then callback(data.Name) return end
    end
    callback("Unknown song")
end

local function loadMusicWithPreload(soundId, autoPlay, callback)
    if not musicSound then createMusicSound() end
    musicSound:Stop()
    musicSound.SoundId = "rbxassetid://" .. soundId
    _G.CurrentMusicId = soundId
    musicPlaying = autoPlay
    local success, err = pcall(function() ContentProvider:PreloadAsync({musicSound}) end)
    if not success then if callback then callback(false) end return end
    local loaded = false
    local conn = musicSound.Loaded:Connect(function() loaded = true; conn:Disconnect() end)
    local timeout = 5
    local start = tick()
    while not loaded and tick() - start < timeout do task.wait(0.1) end
    if not loaded then if callback then callback(false) end return end
    if autoPlay then
        musicSound:Play()
        musicPlaying = true
        notify("Music", "Now playing", 3, false)
    end
    if callback then callback(true) end
end

local function musicStabilityLoop()
    while true do
        task.wait(5)
        if musicPlaying and musicSound and not musicSound.IsPlaying and _G.CurrentMusicId then
            pcall(function() musicSound:Play(); notify("Music", "Resumed playback", 2, false) end)
        end
    end
end
task.spawn(musicStabilityLoop)

local function runPoolESP()
    local cam = Camera if not cam then return end
    local camPos = cam.CFrame.Position
    local screenSize = cam.ViewportSize
    local targets = {}
    for tblName, tbl in pairs({survivors=espData.survivors, killers=espData.killers, generators=espData.generators, batteries=espData.batteries, fuses=espData.fuses}) do
        for obj, colorInfo in pairs(tbl) do
            if obj and obj.Parent then
                local bPos = (obj:IsA("Model") and obj:GetPivot().Position) or (obj:IsA("BasePart") and obj.Position)
                if bPos then
                    local sPos, onScreen = cam:WorldToViewportPoint(bPos)
                    local dist = (camPos - bPos).Magnitude
                    local isRelevant = (dist < 250) or (onScreen or (sPos.X > -1000 and sPos.X < screenSize.X+1000) and (sPos.Y > -1000 and sPos.Y < screenSize.Y+1000))
                    if isRelevant and dist < 5000 then
                        local priority = (tblName=="killers" and 0) or (tblName=="survivors" and 10) or (tblName=="generators" and 200) or (tblName=="batteries" and 300) or (tblName=="fuses" and 400) or 1000
                        table.insert(targets, {obj=obj, dist=dist, prio=priority, fill=colorInfo.fill, outline=colorInfo.outline or colorInfo.fill})
                    end
                else tbl[obj]=nil end
            end
        end
    end
    table.sort(targets, function(a,b) if a.prio~=b.prio then return a.prio<b.prio end return a.dist<b.dist end)
    for i=1,64 do
        local h = espData.pool[i]
        local target = targets[i]
        if target then
            if target.obj:IsA("Model") and target.prio==10 then
                local c = getSurvivorColor(target.obj)
                h.FillColor, h.OutlineColor = c, c
            else
                h.FillColor, h.OutlineColor = target.fill, target.outline
            end
            h.Adornee = target.obj
        else h.Adornee = nil end
    end
end

local function getRoot(c) return c:FindFirstChild('HumanoidRootPart') or c:FindFirstChild('Torso') or c:FindFirstChild('UpperTorso') end
local function sFly()
    if flying then return end
    repeat task.wait() until LocalPlayer.Character and getRoot(LocalPlayer.Character) and LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
    repeat task.wait() until LocalPlayer:GetMouse()
    if flyKeyDown then flyKeyDown:Disconnect() end
    if flyKeyUp then flyKeyUp:Disconnect() end
    local T = getRoot(LocalPlayer.Character)
    local C = {F=0,B=0,L=0,R=0,Q=0,E=0}
    local lC = {F=0,B=0,L=0,R=0,Q=0,E=0}
    local S_ = 0
    local function FLY()
        flying = true
        local BG = Instance.new('BodyGyro'); local BV = Instance.new('BodyVelocity')
        BG.P = 9e4; BG.Parent = T; BV.Parent = T
        BG.maxTorque = Vector3.new(9e9,9e9,9e9); BG.cframe = T.CFrame
        BV.velocity = Vector3.new(0,0,0); BV.maxForce = Vector3.new(9e9,9e9,9e9)
        task.spawn(function()
            repeat task.wait()
                if LocalPlayer.Character:FindFirstChildOfClass('Humanoid') then
                    LocalPlayer.Character:FindFirstChildOfClass('Humanoid').PlatformStand = true
                end
                if C.L+C.R~=0 or C.F+C.B~=0 or C.Q+C.E~=0 then S_=50 elseif S_~=0 then S_=0 end
                if (C.L+C.R)~=0 or (C.F+C.B)~=0 or (C.Q+C.E)~=0 then
                    BV.velocity = ((Camera.CFrame.lookVector*(C.F+C.B)) + ((Camera.CFrame*CFrame.new(C.L+C.R, (C.F+C.B+C.Q+C.E)*0.2,0).p)-Camera.CFrame.p))*S_*flySpeed
                    lC = {F=C.F,B=C.B,L=C.L,R=C.R}
                elseif S_~=0 then
                    BV.velocity = ((Camera.CFrame.lookVector*(lC.F+lC.B)) + ((Camera.CFrame*CFrame.new(lC.L+lC.R, (lC.F+lC.B+C.Q+C.E)*0.2,0).p)-Camera.CFrame.p))*S_*flySpeed
                else
                    BV.velocity = Vector3.new(0,0,0)
                end
                BG.cframe = Camera.CFrame
            until not flying
            C = {F=0,B=0,L=0,R=0,Q=0,E=0}; lC = {F=0,B=0,L=0,R=0,Q=0,E=0}; S_=0
            BG:Destroy(); BV:Destroy()
            if LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass('Humanoid') then
                LocalPlayer.Character:FindFirstChildOfClass('Humanoid').PlatformStand = false
            end
        end)
    end
    local mouse = LocalPlayer:GetMouse()
    flyKeyDown = mouse.KeyDown:Connect(function(k)
        k = k:lower()
        if k=='w' then C.F=1 elseif k=='s' then C.B=-1 elseif k=='a' then C.L=-1 elseif k=='d' then C.R=1
        elseif k=='e' then C.Q=1 elseif k=='q' then C.E=-1 end
    end)
    flyKeyUp = mouse.KeyUp:Connect(function(k)
        k = k:lower()
        if k=='w' then C.F=0 elseif k=='s' then C.B=0 elseif k=='a' then C.L=0 elseif k=='d' then C.R=0
        elseif k=='e' then C.Q=0 elseif k=='q' then C.E=-0 end
    end)
    FLY()
end

local function setupParryDetection(killer)
    local function detect(desc)
        if desc:IsA("Highlight") and desc.Name:lower():find("attack") then
            local conn = desc:GetPropertyChangedSignal("FillColor"):Connect(function()
                local r,g,b = desc.FillColor.R*255, desc.FillColor.G*255, desc.FillColor.B*255
                if r>250 and g<10 and b<10 then tryParry(killer) end
            end)
            table.insert(parryConns, conn)
        elseif desc:IsA("Animator") then
            local conn = desc.AnimationPlayed:Connect(function(track)
                local name = track.Animation and track.Animation.Name:lower() or ""
                if name:find("swing") or name:find("slash") or name:find("attack") then tryParry(killer) end
            end)
            table.insert(parryConns, conn)
        end
    end
    for _, d in ipairs(killer:GetDescendants()) do detect(d) end
    table.insert(parryConns, killer.DescendantAdded:Connect(detect))
end

    if noclipConn then return end
    noclipConn = RunService.Stepped:Connect(function()
        if noclipEnabled then
            applyNoclip()
        end
    end)
    applyNoclip()
end

local function stopNoclip()
    if noclipConn then
        noclipConn:Disconnect()
        noclipConn = nil
    end
    restoreCollision()
end
